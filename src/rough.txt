import java.time.LocalDate;
import java.time.format.DateTimeFormatter;
import java.util.Date;

public class DateUtil {

    public static String getYearMonthFromDate(Date date) {
        LocalDate localDate = date.toInstant().atZone(ZoneId.systemDefault()).toLocalDate();
        return localDate.format(DateTimeFormatter.ofPattern("dd-MMM"));
    }
}
Date date = new Date(); // Your Date object
String yearMonth = DateUtil.getYearMonthFromDate(date);
System.out.println(yearMonth); // Output: 07-Jul
